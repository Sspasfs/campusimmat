\hypertarget{md_public_2glpi_2vendor_2wapmorgan_2unified-archive_2docs_2_drivers_autotoc_md7496}{}\doxysection{\texorpdfstring{Proposals for installation drivers}{Proposals for installation drivers}}\label{md_public_2glpi_2vendor_2wapmorgan_2unified-archive_2docs_2_drivers_autotoc_md7496}

\begin{DoxyItemize}
\item {\bfseries{In common case\+:}}
\begin{DoxyItemize}
\item install {\ttfamily Tar\+By\+Pear}, {\ttfamily Nelexa\+Zip}
\item If installed {\ttfamily 7za} utility -\/ configure {\ttfamily Seven\+Zip} driver
\item Else if installed {\ttfamily tar} or {\ttfamily unzip} utilities -\/ configure {\ttfamily Alchemy\+Zippy} driver
\end{DoxyItemize}
\item {\bfseries{In docker/on VDS\+:}}
\begin{DoxyItemize}
\item install all php extensions ({\ttfamily zip, rar, phar, zlib, bz2})
\item install utility ({\ttfamily 7za} -\/ {\ttfamily p7zip-\/full} on ubuntu) and {\ttfamily Seven\+Zip} driver
\end{DoxyItemize}
